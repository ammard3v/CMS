Imports System.Data.SqlClient

Public Class Dashboard
    Dim connectionString As String = "Data Source=DESKTOP-TV2H6SR;Initial Catalog=student;Integrated Security=True;"
    Dim connection As New SqlConnection(connectionString)

    Private Sub UpdateTotalStudents()
        Try
            connection.Open()
            Dim query As String = "SELECT COUNT(*) FROM STable"
            Dim command As New SqlCommand(query, connection)
            Dim totalStudents As Integer = Convert.ToInt32(command.ExecuteScalar())
            totStudent.Text = totalStudents.ToString()
        Catch ex As Exception
            MessageBox.Show("Error updating total students: " & ex.Message)
        Finally
            connection.Close()
        End Try
    End Sub

    Private Sub UpdateTotalFees()
        Try
            connection.Open()
            Dim oldTotalFeeQuery As String = "SELECT SUM(Sfees) FROM STable"
            Dim oldTotalFeeCommand As New SqlCommand(oldTotalFeeQuery, connection)
            Dim oldTotalFeesResult As Object = oldTotalFeeCommand.ExecuteScalar()
            Dim oldTotalFees As Integer = If(oldTotalFeesResult IsNot DBNull.Value, Convert.ToInt32(oldTotalFeesResult), 0)

            Dim newFeeQuery As String = "SELECT SUM(StAmount) FROM StudentFees"
            Dim newFeeCommand As New SqlCommand(newFeeQuery, connection)
            Dim newFeeResult As Object = newFeeCommand.ExecuteScalar()
            Dim newFee As Integer = If(newFeeResult IsNot DBNull.Value, Convert.ToInt32(newFeeResult), 0)

            Dim updatedTotalFees As Integer = oldTotalFees + newFee


            totFee.Text = updatedTotalFees.ToString()

        Catch ex As Exception
            MessageBox.Show("Error updating total fees: " & ex.Message)
        Finally
            connection.Close()
        End Try
    End Sub

    Public Sub UpdateDashboard()
        UpdateTotalStudents()
        UpdateTotalFees()
    End Sub

    Private Sub Dashboard_Load(sender As Object, e As EventArgs) Handles MyBase.Load
        UpdateDashboard()
    End Sub

    Private Sub Button3_Click(sender As Object, e As EventArgs) Handles Button3.Click
        Dim StudentForm As New Student()
        StudentForm.Show()
        Me.Hide()
    End Sub

    Private Sub Button2_Click(sender As Object, e As EventArgs) Handles Button2.Click
        Dim FeesForm As New Fees
        FeesForm.Show()
        Me.Hide()
    End Sub

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim result = MessageBox.Show("Are you sure ?", "Confirmation", MessageBoxButtons.YesNo)
        If result = DialogResult.Yes Then
            Dim LoginForm As New Login()
            LoginForm.Show()
            Me.Hide()
            Application.Exit()
        End If

    End Sub

End Class
